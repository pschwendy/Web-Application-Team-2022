CREATE TABLE users(pk SERIAL PRIMARY KEY,
email VARCHAR(50) NOT NULL,
password VARCHAR(50) NOT NULL,
name VARCHAR(50) NOT NULL,
lastLogin TIMESTAMP);

CREATE TABLE reservations(pk SERIAL PRIMARY KEY,
userID INTEGER NOT NULL,
isSeat BOOLEAN NOT NULL,
seatID INTEGER,
event VARCHAR(50),
eventID INTEGER,
timestamp TIMESTAMP NOT NULL);

CREATE TABLE furniture(pk SERIAL PRIMARY KEY,
isSeat BOOLEAN,
width INTEGER,
height INTEGER,
x INTEGER NOT NULL,
y INTEGER NOT NULL,
available BOOLEAN,
roomID INTEGER NOT NULL);

CREATE TABLE rooms(pk SERIAL PRIMARY KEY,
name VARCHAR(50) NOT NULL);

CREATE TABLE events(pk SERIAL PRIMARY KEY,
name VARCHAR(50) NOT NULL,
description VARCHAR(500) NOT NULL,
capacity INTEGER NOT NULL, 
image VARCHAR(100) NOT NULL);
